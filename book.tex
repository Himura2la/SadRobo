	\ifdim\Gin@nat@width>\linewidth
		\linewidth
	\else
		\Gin@nat@width
	\fi}

\newcommand{\newdialog}[3]{
	\chapter{Робот, которому грустно, и #2}
		\input{#2}
		\begin{figure}[p!]
			\centering
			\includegraphics[height = 190mm, width = \linewidth, keepaspectratio]{#2}    %170 для того, чтобы вместилась надпись
			\settoheight\imageheight{\includegraphics{#2}}
			\settowidth\imagewidth{\includegraphics{#2}}
			\ifthenelse{\imagewidth > \imageheight}{\center #3}{}
		\end{figure}
	\setcounter{dianum}{#1}
}
